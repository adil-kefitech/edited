pythonclass ActiveCampsList(APIView):
    """
    Returns all active camps (sub_camp + camp) for the currently
    logged-in Camp Officer so the user can select which camp
    the bundle should be allocated to.
    """
    def post(self, request):
        try:
            user = request.user
            camp_allocations = CampRoleAllocation.objects.filter(
                camp_user=user,
                camp_group__group__name__in=[CAMP_OFFICER],
                status_id=ACTIVE_STATE,
            ).select_related(
                "camp_schedule__sub_camp__camp"
            )

            camps_data = []
            seen = set()
            for alloc in camp_allocations:
                sub_camp = alloc.camp_schedule.sub_camp
                camp = sub_camp.camp
                key = (alloc.id, sub_camp.id)
                if key not in seen:
                    seen.add(key)
                    camps_data.append({
                        "camp_role_allo_id": alloc.id,     # used as selection key
                        "sub_camp_id": sub_camp.id,
                        "sub_camp_name": sub_camp.name,    # adjust field name to match your model
                        "camp_id": camp.id,
                        "camp_name": camp.name,            # adjust field name to match your model
                    })

            return format_response(
                True,
                FETCH_SUCCESS_MSG,
                data={"camps": camps_data},
                status_code=status.HTTP_200_OK,
                template_name="answer_sheet_distribution_online.html",
            )
        except Exception as e:
            logger.error(e, exc_info=True)
            return format_response(
                False,
                BAD_GATEWAY,
                {},
                BAD_GATEWAY_ERROR_CODE,
                status_code=status.HTTP_400_BAD_REQUEST,
            )
